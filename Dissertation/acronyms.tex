\chapter*{Список сокращений}             % Заголовок
\addcontentsline{toc}{chapter}{Список сокращений}  % Добавляем его в оглавление
\printnomenclature[3.5cm] % Значение ширины столбца с обозначениями стоит подбирать вручную

\begin{description}
	
	\item[IoT] (Internet of Things) — Интернет вещей;
	
	\item[Edge Computing] — периферийные вычисления;
	
	\item[Fog Computing] — туманные вычисления;
	
	\item[MEC] (Multi-access edge computing) — периферийные вычисления с множественным доступом;
	
	\item[MAS] (Multi-Agent Systems) — многоагентные системы;
	
	\item[MARL] (Multi-Agent Reinforcement Learning) — многоагентное обучение с подкреплением;
	
	\item[RL] (Reinforcement Learning) — обучение с подкреплением;
	
	\item[MDP] (Markov Decision Process) — марковский процесс принятия решений;
	
	\item[Dec-POMDP] (Decentralized Partially Observable Markov Decision Process) — децентрализованный частично наблюдаемый марковский процесс принятия решений;
	
	\item[QMIX] — алгоритм факторизации функции ценности в многоагентном обучении;
	
	\item[VCG] (Vickrey–Clarke–Groves) — механизм Викри–Кларка–Гровса;
	
	\item[DSIC] (Dominant Strategy Incentive Compatibility) — доминантная стратегическая совместимость;
	
	\item[CTDE] (Centralized Training with Decentralized Execution) — централизованное обучение с децентрализованным исполнением;
	
	\item[NN] (Neural Network) — нейронная сеть;
	
	\item[GRU] (Gated Recurrent Unit) — рекуррентная нейронная сеть с управляемыми вентилями;
	
	\item[DQN] (Deep Q-Network) — глубокая Q-сеть;
	
	\item[CPU] (Central Processing Unit) — центральный процессор;
	
	\item[GPU] (Graphics Processing Unit) — графический процессор;
	
	\item[ITS] (Intelligent Transport Systems) — интеллектуальные транспортные системы;
	
	\item[AMR] (Autonomous Mobile Robot) — автономный мобильный робот;
	
	\item[FMS] (Flexible Manufacturing Systemt) — гибкая производственная система;
	
	\item[QoS] (Quality of Service) — качество обслуживания;
	
	\item[SLA] (Service-level agreement) — соглашение об уровне предоставления услуги;
	
\end{description}
